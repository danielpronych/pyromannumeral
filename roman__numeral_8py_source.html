33 # @brief Integer to Roman Numeral Conversion Routine
34 # @param integer The integer number sent for conversion.
35 # @exception ValueError will be thrown if an invalid integer specified.
36 # @return string representing the Roman Numeral equivalent of input integer.
37 def integer_to_roman(integer):
38  try:
39  number = int(integer)
40  except:
41  raise ValueError("Input not a valid Integer: %s." % str(integer))
42  # This instance does not support Roman Numerals >= 4 million
43  if(number > 3999999):
44  raise ValueError("Input not a valid Integer: %s." % str(integer))
45  result = ""
46  for v, n in sorted(__numerals.items(), reverse=True):
47  while number >= v:
48  result += n
49  number -= v
50  return result

54 # @brief Roman Numeral to Integer Conversion Routine
55 # @param numeral The Roman Numeral string sent for conversion.
56 # @exception ValueError will be thrown for an invalid Roman Numeral.
57 # @return string representing the Roman Numeral equivalent of input numeral.
58 #
59 def roman_to_integer(numeral):
60  try:
61  # Ensure an uppercase version of the input string is used
62  roman = str(numeral).upper()
63  except Exception, e:
64  raise ValueError("Input not a valid Integer: %s." % str(numeral))
65  result = 0
66  for v, n in sorted(__numerals.items(), reverse=True):
67  my_len = len(n)
68  while n == roman[:my_len]:
69  result += int(v)
70  roman = roman[my_len:]
71  # This will error if one item couldn't be converted, or no conversion match
72  if len(roman) or integer_to_roman(result) != numeral.upper():
73  raise ValueError("Input not a valid Integer: %s." % str(numeral))
74  return result
